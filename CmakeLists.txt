cmake_minimum_required(VERSION 3.15)
project(feature_store)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 寻找依赖（vcpkg 会自动拦截并接管这里的查找逻辑）
find_package(Arrow REQUIRED)
find_package(RocksDB REQUIRED)
include(CTest)

# 启用 Address Sanitizer
option(ENABLE_ASAN "Enable Address Sanitizer" OFF)
if(ENABLE_ASAN)
    if(CMAKE_CXX_COMPILER_ID MATCHES "Clang|AppleClang|GNU")
    add_compile_options(-fsanitize=address -fno-omit-frame-pointer -g)
    add_link_options(-fsanitize=address)
    endif()
endif()


# 添加可执行文件
add_executable(feature_store main.cpp)

add_library(feature_store_engine
    src/engine/frame_codec.cc
    src/engine/engine_config.cc
)

target_include_directories(feature_store_engine
    PUBLIC
    ${PROJECT_SOURCE_DIR}/src/engine
    ${PROJECT_SOURCE_DIR}/LogSystem
)

target_link_libraries(feature_store_engine
    PUBLIC
    Arrow::arrow_static
)

# 链接库
target_link_libraries(feature_store
    PRIVATE
    feature_store_engine
    Arrow::arrow_static
    RocksDB::rocksdb
)

target_include_directories(feature_store PRIVATE
    ${PROJECT_SOURCE_DIR}/LogSystem
)

# 测试代码
include(${PROJECT_SOURCE_DIR}/src/tests/CMakeLists.txt)
include(${PROJECT_SOURCE_DIR}/src/tests/benchmark/CMakeLists.txt)
